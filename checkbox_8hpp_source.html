23 template <template <typename> typename Layout_t>
24 class Checkbox : public Label<Layout_t> {
25  private:
26  using Base_t = Label<Layout_t>;
27 
28  public:
29  enum class State : bool { Unchecked, Checked };
30 
31  struct Display {
32  Glyph_string checked;
33  Glyph_string unchecked;
34  };
35 
36  struct Parameters {
37  State initial_state;
38  Display display;
39  bool locked;
40  };
41 
42  public:
44  sl::Signal<void()> checked;
45 
47  sl::Signal<void()> unchecked;
48 
50  sl::Signal<void()> toggled;
51 
52  public:
54  explicit Checkbox(State initial_state, Display display, bool locked)
55  : Base_t{initial_state == State::Unchecked ? display.unchecked
56  : display.checked,
57  Align::Left, 0, 0, Growth::Dynamic},
58  state_{initial_state},
59  display_{display},
60  locked_{locked}
61  {
62  if (locked_) {
63  display_.checked | Trait::Dim;
64  display_.unchecked | Trait::Dim;
65  this->update_display();
66  }
67  }
68 
70  explicit Checkbox(Parameters p)
71  : Checkbox{p.initial_state, p.display, p.locked} {};
72 
73  public:
75  void check()
76  {
77  if (locked_)
78  return;
79  if (state_ == State::Unchecked) {
80  state_ = State::Checked;
81  this->Base_t::set_text(display_.checked);
82  checked.emit();
83  toggled.emit();
84  }
85  }
86 
88  void uncheck()
89  {
90  if (locked_)
91  return;
92  if (state_ == State::Checked) {
93  state_ = State::Unchecked;
94  this->Base_t::set_text(display_.unchecked);
95  unchecked.emit();
96  toggled.emit();
97  }
98  }
99 
101  void toggle()
102  {
103  switch (state_) {
104  case State::Checked: this->uncheck(); break;
105  case State::Unchecked: this->check(); break;
106  }
107  }
108 
110  auto get_state() const -> State { return state_; }
111 
113  void lock()
114  {
115  locked_ = true;
116  display_.checked | Trait::Dim;
117  display_.unchecked | Trait::Dim;
118  this->update_display();
119  }
120 
122  void unlock()
123  {
124  locked_ = false;
125  display_.checked.remove_traits(Trait::Dim);
126  display_.unchecked.remove_traits(Trait::Dim);
127  this->update_display();
128  }
129 
131  auto is_locked() const -> bool { return locked_; }
132 
134  void set_display(Display d)
135  {
136  display_ = std::move(d);
137  if (locked_) {
138  display_.checked | Trait::Dim;
139  display_.unchecked | Trait::Dim;
140  }
141  this->update_display();
142  }
143 
145  auto get_display() -> Display
146  {
147  auto result = display_;
148  if (locked_) {
149  result.checked.remove_traits(Trait::Dim);
150  result.unchecked.remove_traits(Trait::Dim);
151  }
152  return result;
153  }
154 
155  protected:
156  auto mouse_press_event(Mouse const& m) -> bool override
157  {
158  if (m.button == Mouse::Button::Left)
159  this->toggle();
160  return Base_t::mouse_press_event(m);
161  }
162 
163  private:
164  State state_;
165  Display display_;
166  bool locked_;
167 
168  private:
170  void update_display()
171  {
172  this->Base_t::set_text(state_ == State::Unchecked ? display_.unchecked
173  : display_.checked);
174  }
175 };
