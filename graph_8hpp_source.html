141  void initialize_bitmap(Coordinates c)
142  {
143  auto const visual_width = this->width();
144  auto const visual_height = this->height();
145  if (visual_width == 0 or visual_height == 0)
146  return;
147  auto const horizontal_offset = [this, &c, visual_width] {
148  auto const domain_width = distance(boundary_.west, boundary_.east);
149  auto const interval = domain_width / visual_width;
150  // TODO make interval (h/v) a class member updated on resize or
151  // boundary change.
152  if (interval == 0)
153  return Number_t{0};
154  return distance(boundary_.west, c.x) / interval;
155  }();
156  auto const vertical_offset = [this, &c, visual_height] {
157  auto const domain_height =
158  distance(boundary_.north, boundary_.south);
159  auto const interval = domain_height / visual_height;
160  if (interval == 0)
161  return Number_t{0};
162  auto const distance_from_bottom =
163  distance(boundary_.north, c.y) / interval;
164  return visual_height - distance_from_bottom; // Invert
165  }();
166  auto& bitmap = map_[{static_cast<std::size_t>(horizontal_offset),
167  static_cast<std::size_t>(vertical_offset)}];
168 
169  auto const h_dec = std::fmod(horizontal_offset, 1.);
170  auto const v_dec = std::fmod(vertical_offset, 1.);
171  if (h_dec < 0.5) {
172  if (v_dec < 0.25)
173  bitmap.set_top_left();
174  else if (v_dec < 0.5)
175  bitmap.set_mid_top_left();
176  else if (v_dec < 0.75)
177  bitmap.set_mid_bottom_left();
178  else
